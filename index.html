var titulo = /(Caracteres Especiais)/;

\w - [A-Za-z0-9_]

"abc" /\w/ = "a"

"$10" /\w/ = "1"

"JS_Day" /\w\w\w/ = "JS_"

"JS4Day" /\w\w\w\w/ = "JS4D"

\d - [0-9]

"abc" /\d/ = null

"$10" /\d\d/ = "10"

"a1 a2 a3 a4" /\d\d\d\d/ = null

Como ficaria para pegar apenas os números dessa última sentença?

"a1 a2 a3 a4" /\d/g = "1234"
